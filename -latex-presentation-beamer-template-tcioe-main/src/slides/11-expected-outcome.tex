\begin{frame}{Expected Outcome}
    \begin{itemize}
        \item \textbf{Robust Offline Geolocation:} A system capable of estimating geographic coordinates and heading from a single mountain photo without GPS.
        \item \textbf{Lightweight Processing:} Real-time matching on mobile hardware using FFT and DTW optimizations.
        \item \textbf{Visual Feedback:} An interface that overlays the extracted skyline with the database horizon for user verification.
        \item \textbf{High Accuracy:} Successful identification of the correct $500 \text{ m} \times 500 \text{ m}$ grid cell within the top-$k$ ranked results.
    \end{itemize}
\end{frame}
